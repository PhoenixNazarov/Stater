package fsm.door;

import com.stater.statemachine.java.*;
import java.util.*;

class Door {
    static class DoorStateMachine extends StaterStateMachine<String, EmptyContext>{
        public DoorStateMachine() {
            super(
                List.of(
                    new Transition<>(
                        "preOpen",
                        "CLOSE",
                        "AJAR", 
                        ctx -> true, 
                        ctx -> {}
                    ),
                    new Transition<>(
                        "preClose",
                        "OPEN",
                        "AJAR", 
                        ctx -> true, 
                        ctx -> {}
                    ),
                    new Transition<>(
                        "open",
                        "AJAR",
                        "OPEN", 
                        ctx -> true, 
                        ctx -> {}
                    ),
                    new Transition<>(
                        "close",
                        "AJAR",
                        "CLOSE", 
                        ctx -> true, 
                        ctx -> {}
                    ),
                    new Transition<>(
                        "ajarPlus",
                        "AJAR",
                        "AJAR", 
                        ctx -> true, 
                        ctx -> {}
                    ),
                    new Transition<>(
                        "ajarMinus",
                        "AJAR",
                        "AJAR", 
                        ctx -> true, 
                        ctx -> {}
                    )
                ),
                new EmptyContext(),
                "OPEN"
            );
        }
    
        void preOpen() {
            transition("preOpen");
        }
    
        void preClose() {
            transition("preClose");
        }
    
        void open() {
            transition("open");
        }
    
        void close() {
            transition("close");
        }
    
        void ajarPlus() {
            transition("ajarPlus");
        }
    
        void ajarMinus() {
            transition("ajarMinus");
        }
    }
}
