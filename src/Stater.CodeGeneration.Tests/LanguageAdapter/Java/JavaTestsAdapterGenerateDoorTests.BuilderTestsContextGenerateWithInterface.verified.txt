    @Test
    void testScenario0() {
        var sm = (Door.TypesDoorStateMachine)Door.builderDoorStateMachine.build();
        sm.disableEvents();
    

        sm.preClose();
        assertEquals(sm.getState(), "AJAR");

        sm.close();
        assertEquals(sm.getState(), "CLOSE");

        sm.preOpen();
        assertEquals(sm.getState(), "AJAR");

        sm.ajarPlus();
        assertEquals(sm.getState(), "AJAR");

        sm.open();
        assertEquals(sm.getState(), "OPEN");

    }
